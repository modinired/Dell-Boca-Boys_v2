```python
from core.memory import (
    create_enhanced_memory_manager,
    MemoryProvider,
    MemoryType,
    MemoryQuery

    provider=MemoryProvider.HYBRID  # Uses both Mem0 and CESAR

await memory_manager.initialize()

    sender_id="terry_delmonaco",
    receiver_id="victoria_sterling",
    message_type="strategy_consultation",
    content={
        "topic": "Q4 market analysis",
        "urgency": "high",
        "context": "portfolio_optimization"
    },
    importance=0.9
)

# Store learning data (routes to CESAR for analytics)
learning_id = await memory_manager.store_learning_data(
    agent_id="terry_delmonaco",
    learning_type="pattern_recognition",
    learning_content={
        "pattern": "market_volatility_correlation",
        "accuracy": 0.94,
        "training_samples": 50000
    },
    effectiveness=0.94
)

# Retrieve memories with smart routing
query = MemoryQuery(
    memory_types=[MemoryType.AGENT_COMMUNICATION, MemoryType.LEARNING_DATA],
    agent_filter="terry_delmonaco",
    importance_threshold=0.7,
    limit=20
)
results = await memory_manager.retrieve_memory(query)

# Get performance analytics
analytics = await memory_manager.get_performance_analytics()
print(f"Token reduction: {analytics['token_reduction_pct']:.1f}%")
print(f"Latency improvement: {analytics['latency_improvement_pct']:.1f}%")
```
